001package absimpa;
002
003import java.util.Collections;
004import java.util.EnumSet;
005import java.util.Map;
006
007import absimpa.parserimpl.AbstractParser;
008import absimpa.parserimpl.TokenParser;
009
010/**
011 * <p>a grammar to recognize a single token.</p>
012 * @param <C> is the type of token recognized.
013 * @param <N> is the type of object created from the token
014 */
015public class TokenGrammar<N,C extends Enum<C>>
016    extends Grammar<N,C>
017{
018  private final C code;
019
020  public TokenGrammar(C code) {
021    this.code = code;
022  }
023  @Override
024  protected Iterable<Grammar<N,C>> children()
025  {
026    return Collections.emptyList();
027  }
028  protected AbstractParser<N,C> buildParser(Map<Grammar<N,C>,First<N,C>> firstOf) {
029    return new TokenParser<>(code);
030  }
031  @Override
032  protected First<N,C> computeFirst(Map<Grammar<N,C>,First<N,C>> firstOf)
033  {
034    EnumSet<C> firstSet = EnumSet.of(code);
035    return new First<>(firstSet, false);
036  }
037  public String getDetail() {
038    return code.toString();
039  }
040  public String _ruleString() {
041    String name = getName();
042    String representation = null;
043    representation = code.toString();
044    
045    if( name==null ) {
046      return representation;
047    }
048    return name + "(" + representation +")";
049  }
050}
