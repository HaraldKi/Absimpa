012/**
013 * <p>
014 * is an alternative to {@link GrammarBuilder} to create a grammar by
015 * providing a string syntax resembling <a
016 * href="http://en.wikipedia.org/wiki/Backus%E2%80%93Naur_Form">Backus Naur
017 * Form (BNF)</a> to build recursive grammars. The syntax used, as
018 * automatically generated with {@link Grammar#toBNF()}, is as follows:
019 * </p>
020 * 
021 * <pre>
022 * expr → (product ("[|]" product)*)
023 * product → term+
024 * term → (literal oper?)
025 * literal → ("[A-Za-z][A-Za-z0-9]*" | ("[%]"? "[(]" [expr] "[)]"))
026 * oper → ("[+]" | "[*]" | "[?]" | ("[{]" range "[}]"))
027 * range → ("[0-9]+" ("[,]" upperlimit)?)
028 * upperlimit → ("[0-9]+" | "[*]")
029 * </pre>
030 * <p>
031 * Tokens are represented by their regular expressions and the brackets in
032 * {@code [expr]} merely point out that this is a recursive use of the rule
033 * {@code expr}. The percent sign is used to insert {@link NodeFactory}
034 * objects in the right place.
035 * </p>
036 * <p>
037 * Add rules with {@link #rule rule()} and finally compile your start symbol
038 * with {@link #compile compile()}.
039 * </p>
040 * 
041 * @param <N> type of node
042 * @param <C> type of tokencode
043 */
044public class BNF<N, C extends Enum<C>> {
045  // For grammars entered with rule();
046  private final Map<String,Grammar<N,C>> syntaxMap;
047  
048  // For grammar names not yet entered with rule()
049  private final Map<String,Recurse<N,C>> placeHolders = new HashMap<>();
050
051  // our lexer and parser
052  private final SimpleLexer<Node,TokenCode> lex;
053  private final Parser<Node,TokenCode> parser;
054  
055  // only used during parsing to allow MyLeafs to fetch the next node factory
056  private List<NodeFactory<N>> nfmarkStack = null;
057  
058  /*+******************************************************************/
059  private static enum TokenCode  {
060    EOF(""), 
061    SYMBOL("[A-Za-z][A-Za-z0-9]*"), 
062    NUMBER("[0-9]+"), 
063    STAR("[*]"), 
064    PLUS("[+]"), 
065    OPAREN("[(]"), 
066    CPAREN("[)]"), 
067    QUESTIONMARK("[?]"), 
068    BAR("[|]"), 
069    PERCENT("%"),
070    LBRACE("[{]"), 
071    RBRACE("[}]"), 
072    COMMA(",");
073
074    private final String regex;
075    
076    private TokenCode(String regex) {
077      this.regex = regex;
078    }
079    public String getRegex() {
080      return regex;
081    }
082  }
083  /* +***************************************************************** */
084  /**
085   * <p>creates a BNF parser and registers all elements {@code c} of the
086   * enumeration as tokens of the grammar with {@code new
087   * TokenGrammar<N,C>(c)}. The name of each {@code c} is immediately usable
088   * in rules.</p>
089   */
090  public BNF(Class<C> enumClass) {
091    C[] constants = enumClass.getEnumConstants();
092    syntaxMap = new HashMap<>(constants.length);
093    for(C c : constants) {
094      syntaxMap.put(c.toString(), new TokenGrammar<N,C>(c));
095    }
096
097    Grammar<Node,TokenCode> expr = getGrammar();
098
099    parser = expr.compile();
100
101    lex = new SimpleLexer<>(TokenCode.EOF, new MyLeafs());
102    for(TokenCode c : TokenCode.values()) {
103      if (c==TokenCode.EOF) continue;
104      lex.addToken(c, c.getRegex());
105    }
106    lex.setSkipRe("[\\s]+");
107  }
108  /*+******************************************************************/
109  private Grammar<Node,TokenCode> getGrammar() {
110    // This is what we want to parse as right side of rules
111    // expr -> product ( "|" product)*
112    // product -> term (term)*
113    // term -> literal (oper)?
114    // oper -> "+" | "*" | "?" | "{" range "}"
115    // range -> number ("," (number | "*"))?
116    // literal -> symbol | "%"? "(" expr ")"
117    // symbol -> {any token registered by the user}
118
119    
120    //..
121    // TODO: Require to pre-decleare recursive rules. Otherwise a simple typo
122    // ends up somewhere with an NPE
123    //
124
125    GrammarBuilder<Node,TokenCode> gb = new GrammarBuilder<>(null);
126
127    Recurse<Node,TokenCode> recurse = new Recurse<>();
128
129    Grammar<Node,TokenCode> symbol = gb.token(TokenCode.SYMBOL);
130    Grammar<Node,TokenCode> number = gb.token(TokenCode.NUMBER);
131    Grammar<Node,TokenCode> oparen = gb.token(TokenCode.OPAREN);
132    Grammar<Node,TokenCode> cparen = gb.token(TokenCode.CPAREN);
133    Grammar<Node,TokenCode> star = gb.token(TokenCode.STAR);
134    Grammar<Node,TokenCode> plus = gb.token(TokenCode.PLUS);
135    Grammar<Node,TokenCode> questionmark = gb.token(TokenCode.QUESTIONMARK);
136    Grammar<Node,TokenCode> bar = gb.token(TokenCode.BAR);
137    Grammar<Node,TokenCode> nfmark = gb.token(TokenCode.PERCENT);
138    Grammar<Node,TokenCode> lbrace = gb.token(TokenCode.LBRACE);
139    Grammar<Node,TokenCode> rbrace = gb.token(TokenCode.RBRACE);
140    Grammar<Node,TokenCode> comma = gb.token(TokenCode.COMMA);
141
142    Grammar<Node,TokenCode> parenExpr =
143        gb.seq(gb.opt(nfmark), oparen, recurse, cparen)
144          .setNodeFactory(makeNodeFactory);
145
146    Grammar<Node,TokenCode> literal = 
147      gb.choice(symbol, parenExpr)
148      .setName("literal");
149
150    Grammar<Node,TokenCode> range = 
151      gb.seq(number,
152             gb.seq(comma, 
153                    gb.choice(number, star).setName("upperlimit")).opt()).setName("range");
154                                      
155    Grammar<Node,TokenCode> oper = 
156      gb.choice(plus, 
157                star, 
158                questionmark, 
159                gb.seq(lbrace, range, rbrace)
160                ).setName("oper");
161
162    Grammar<Node,TokenCode> term =
163        gb.seq(literal, gb.opt(oper))
164        .setNodeFactory(makeLiteral)
165        .setName("term");
166
167    Grammar<Node,TokenCode> product =
168        gb.repeat(term, 1, Integer.MAX_VALUE)
169        .setNodeFactory(makeProduct)
170        .setName("product");
171
172    Grammar<Node,TokenCode> expr =
173        gb.seq(product, gb.seq(bar, product).star())
174        .setNodeFactory(makeOr)
175        .setName("expr");
176
177    recurse.setChild(expr);
178    //recurse.setName("r#expr");
179    return expr;
180  }
181  /*+******************************************************************/
182  private final NodeFactory<Node> makeNodeFactory = new NodeFactory<Node>() {
183    @Override
184    public Node create(List<Node> children) {
185      if( children.size()==1 ) {
186        return children.get(0);
187      }
188      assert children.size()==2 : "must have one or two children but have "+children.size(); 
189      assert children.get(0).nodeFactory != null;
190      
191      Node childNode = children.get(1);
192      Grammar<N,C> child = childNode.getGrammar();
193      NodeFactory<N> nf = children.get(0).getNodeFactory();
194      child.setNodeFactory(nf);
195      return childNode;
196    }
197    public String toString() {
198      return "";
199    }
200  };
201  /*+******************************************************************/
202  private final NodeFactory<Node> makeLiteral= new NodeFactory<Node>() {
203    @Override
204    public Node create(List<Node> children) {
205      if( children.size()==1 ) {
206        return children.get(0);
207      }
208
209      if( children.get(1).getNum()!=null ) {
210        return createRepeatFromRange(children);
211      }
212
213      Grammar<N,C> child = children.get(0).getGrammar();
214      switch( children.get(1).getCode() ) {
215      case STAR:
216        return new Node(new Repeat<>(0, Integer.MAX_VALUE, child));
217      case QUESTIONMARK:
218        return new Node(new Repeat<>(0, 1, child));
219      case PLUS:
220        return new Node(new Repeat<>(1, Integer.MAX_VALUE, child));
221      default:
222        assert false : "should not have TokenCode "+children.get(1).getCode();
223      }
224      return null;
225    }
226    public String toString() {
227      return "";
228    }
229    private final Node createRepeatFromRange(List<Node> children) {
230      int from = children.get(1).getNum().intValue();
231      int to;
232      if( children.size()==3 ) {
233        if( children.get(2).getCode()==TokenCode.STAR ) {
234          to = Integer.MAX_VALUE;          
235        } else {
236          to = children.get(2).getNum().intValue();
237        }
238      } else {
239        to = from;
240      }
241      Grammar<N,C> g = new Repeat<>(from, to, children.get(0).getGrammar());
242      return new Node(g);
243    }
244  };
245  /*+******************************************************************/
246  private final NodeFactory<Node> makeProduct = new NodeFactory<Node>() {
247    @Override
248    public Node create(List<Node> children) {
249      if( children.size()==1 ) {
250        return children.get(0);
251      }
252      Sequence<N,C> result = new Sequence<>(children.get(0).getGrammar());
253      for(int i=1; i<children.size(); i++) {
254        result.add(children.get(i).getGrammar());
255      }
256      return new Node(result);
257    }
258    public String toString() {
259      return "";
260    }
261  };
262  /*+******************************************************************/
263  private final NodeFactory<Node> makeOr = new NodeFactory<Node>() {
264    @Override
265    public Node create(List<Node> children) {
266      if( children.size()==1 ) {
267        return children.get(0);
268      }
269      Choice<N,C> result = new Choice<>(children.get(0).getGrammar());
270      for(int i=1; i<children.size(); i++) {
271        result.or(children.get(i).getGrammar());
272      }
273      return new Node(result);
274    }
275    public String toString() {
276      return "";
277    }
278  };

290  /**
291   * <p>
292   * create an empty rule to be filled later. This may be necessary as a
293   * forward declaration for recursive grammar symbol use.
294   * </p>
295   */
296  public Grammar<N,C> rule(String name) {
297    if( syntaxMap.containsKey(name) || placeHolders.containsKey(name) ) {
298      throw new IllegalArgumentException(name+" already defined.");
299    }
300    
301    Recurse<N,C> g = new Recurse<>();
302    placeHolders.put(name, g);
303    return g;
304  }

307   * <p>
308   * same as the three parameter {@link #rule(String,String,NodeFactory)
309   * rule()}, but without any node factory.</p>

311  public Grammar<N,C> rule(String name, String expansion) throws ParseException {
312    return rule(name, expansion, Collections.<NodeFactory<N>>emptyList()); 
313  }

315  /**
316   * <p>translates a rule with the given {@code name} into a grammar described
317   * by {@code expansion}, using the given node factory. An example expansion
318   * according to the grammar desribed above is:</p>
319   * 
320   * <pre>
321   * %(term (PLUS term)*)
322   * </pre>
323   *
324   * <p>
325   * This assumes that {@code term} is already defined as a rule, at least
326   * with {@link #rule(String)}. The same holds for {@code PLUS}. Most likely
327   * the latter is defined already by the constructor, if {@code PLUS} is
328   * part of the token code enumeration {@code <C>}.
329   * </p>
330   * <p>
331   * The node factory provided is inserted at the place of the percent sign.
332   * In this example, the node factory would need to be able to combine one
333   * or more objects of type {@code N} representing {@code term} elements
334   * into one new {@code N}.</p>
335   */
336  public Grammar<N,C> rule(String name, String expansion, NodeFactory<N> nf) throws ParseException {
337    List<NodeFactory<N>> l = new ArrayList<>(1);
338    l.add(nf);
339    return rule(name, expansion, l);
340  }

342  /**
343   * <p>create a rule that uses two node factories.</p>
344   * @see #rule(String,String,NodeFactory)
345   */
346  public Grammar<N,C> rule(String name, String expansion, 
347                   NodeFactory<N> nf1, NodeFactory<N> nf2) throws ParseException {
348    List<NodeFactory<N>> l = new ArrayList<>(2);
349    l.add(nf1);
350    l.add(nf2);
351    return rule(name, expansion, l);
352  }
353  /*+******************************************************************/
354  /**
355   * <p>create a rule that uses many node factories.</p>
356   * @see #rule(String,String,NodeFactory)
357   */
358  public Grammar<N,C> rule(String name, String expansion, 
359                   NodeFactory<N> nf1, 
360                   NodeFactory<N> nf2,
361                   NodeFactory<N> ...more) throws ParseException {
362    List<NodeFactory<N>> l = new ArrayList<>(2+more.length);
363    l.add(nf1);
364    l.add(nf2);
365    for(NodeFactory<N> nf : more) l.add(nf);
366    return rule(name, expansion, l);
367  }
368  /* +***************************************************************** */
369  private Grammar<N,C> rule(String name, String expansion,
370                   List<NodeFactory<N>> nodeFactories) throws ParseException
371  {
372    if( syntaxMap.containsKey(name) ) {
373      throw new IllegalArgumentException(name+" already defined.");
374    }
375    nfmarkStack = nodeFactories;
376    
377    lex.initAnalysis(expansion);
378    Node node = parser.parse(lex);
379
380    Grammar<N,C> g = node.getGrammar();
381    g.setName(name);
382
383    Recurse<N,C> placeHolder = placeHolders.remove(name);
384    if( placeHolder!=null ) {
385      placeHolder.setChild(g);
386      placeHolder.setName("r#"+name);
387      g = placeHolder;
388    }
389
390    syntaxMap.put(name, g);
391
392    return g;
393  }
394  /*+******************************************************************/
395  /**
396   * <p>
397   * compiles a previously defined rule. This will typically be the start
398   * symbol of your grammar.
399   * </p>
400   * 
401   * @throws IllegalStateException if there are rules registered with
402   *         {@link #rule(String)} that have no expansion yet. You can always
403   *         circumvent this restriction by using {@link #getGrammar} and
404   *         calling compile on the result. This may, however, throw other
405   *         exceptions.
406   */
407  public Parser<N,C> compile(String ruleNname) {
408    Grammar<N,C> g = syntaxMap.get(ruleNname);
409    if( placeHolders.size()>0 ) {
410      StringBuilder sb = new StringBuilder();
411      sb.append("the following names are undefined:");
412      for(String name : placeHolders.keySet()) {
413        sb.append(' ').append(name);
414      }
415      throw new IllegalStateException(sb.toString());
416    }
417    return  g.compile();
418  }

420  private class MyLeafs implements LeafFactory<Node,TokenCode> {
421    @Override
422    public Node create(SimpleLexer<Node,TokenCode> alex) throws ParseException {
423      TokenCode current = alex.current();
424      
425      switch( current ) {
426      case SYMBOL:
427        String symbolName = alex.currentText();
428        Grammar<N,C> g = syntaxMap.get(symbolName);
429        if( g!=null ) return new Node(g);
430        g = placeHolders.get(symbolName);
431        if( g!=null ) return new Node(g);
432        ParseException pex = alex.parseException(Collections.<TokenCode>emptySet());
433        pex.setMoreInfo("undefined grammar element");
434        throw pex;
435      case STAR:
436      case PLUS:
437      case QUESTIONMARK:
438        return new Node(current);
439      case NUMBER:
440        try {
441          Integer num = Integer.parseInt(alex.currentText());
442          return new Node(num);
443        } catch( NumberFormatException e) {
444          ParseException ex = alex.parseException(Collections.<TokenCode>emptySet());
445          ex.setMoreInfo("cannot parse number, it is likely too large");
446          ex.initCause(e);
447          throw ex;
448        }
449      case PERCENT:
450        if( nfmarkStack.size()<1 ) {
451          ParseException e = alex.parseException(Collections.<TokenCode>emptySet());
452          e.setMoreInfo("not enough node factories");
453          throw e;
454        }
455        return new Node(nfmarkStack.remove(0));
456      default:
457        return null;
458      }
459    }    
460  }

462  private class Node {
463    private final Grammar<N,C> grammar;
464    private final TokenCode code;
465    private final NodeFactory<N> nodeFactory;
466    private final Integer num;
467    public Node(Grammar<N,C> grammar) {
468      this.grammar = grammar;
469      this.code = null;
470      this.nodeFactory = null;
471      this.num = null;
472    }
473    public Node(TokenCode code) {
474      this.grammar = null;
475      this.code = code;
476      this.nodeFactory = null;
477      this.num = null;
478    }
479    public Node(NodeFactory<N> nodeFactory) {
480      this.grammar = null;
481      this.code = null;
482      this.nodeFactory = nodeFactory;
483      this.num= null;
484    }
485    public Node(int num) {
486      this.grammar = null;
487      this.code = null;
488      this.nodeFactory = null;
489      this.num = Integer.valueOf(num);
490    }
491    public Grammar<N,C> getGrammar() {
492      return grammar;
493    }
494    public TokenCode getCode() {
495      return code;
496    }
497    public NodeFactory<N> getNodeFactory() {
498      return nodeFactory;
499    }
500    public Integer getNum() {
501      return num;
502    }
503  }
