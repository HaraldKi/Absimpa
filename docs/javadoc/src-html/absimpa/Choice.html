001package absimpa;
002import java.util.*;
003
004import absimpa.parserimpl.AbstractParser;
005import absimpa.parserimpl.ChoiceParser;
006
007
008public class Choice<N,C extends Enum<C>>
009    extends Grammar<N,C>
010{
011  private final List<Grammar<N,C>> children;
012  // private final Grammar<N,C> firstOptional;
013
014  public Choice(Grammar<N,C> g) {
015    children = new ArrayList<>();
016    children.add(g);
017  }
018  /* +***************************************************************** */
019  public Choice<N,C> or(Grammar<N,C> g) {
020    children.add(g);
021    return this;
022  }
023  /* +***************************************************************** */
024  protected Iterable<Grammar<N,C>> children() {
025    return Collections.unmodifiableList(children);
026  }
027  /* +***************************************************************** */
028  protected String getDetail() {
029    return "";
030  }
031  /* +***************************************************************** */
032  protected AbstractParser<N,C> 
033  buildParser(Map<Grammar<N,C>,First<N,C>> firstOf)
034  {
035    List<AbstractParser<N,C>> childParsers = new ArrayList<>(children.size());
036
037    for(Grammar<N,C> g : children) {
038      childParsers.add(g.build(firstOf));
039    }
040    First<N,C> myFirst = first(firstOf);
041    return new ChoiceParser<>(childParsers, myFirst.firstSet(),
042        myFirst.epsilon);
043  }
044  /* +***************************************************************** */
045  protected First<N,C> computeFirst(Map<Grammar<N,C>,First<N,C>> firstOf) {
046    Grammar<N,C> childGrammar = children.get(0);
047    First<N,C> childFirst = childGrammar.first(firstOf);
048    
049    boolean optional = childFirst.epsilon;
050    EnumSet<C> firstSet = childFirst.firstSet();
051    for(int i=1; i<children.size(); i++) {
052      childGrammar = children.get(i);
053      childFirst = childGrammar.first(firstOf);
054      firstSet.addAll(childFirst.firstSet());
055      optional |= childFirst.epsilon;
056    }
057    return new First<> (firstSet, optional);
058  }
059  /* +***************************************************************** */
060  @Override
061  public String _ruleString() {
062    if( children.size()>1 ) {
063      return Util.join(children, "(", " | ", ")");
064    } else if(children.size()==1) {
065      return children.get(0).toString();
066    } else {
067      return "";
068    }
069  }
070}
