001package absimpa;
002
003/**
004 * <p>
005 * parses input in the form of token codes as provided by a lexical analyzer
006 * and constructs result objects. To obtain a parser, use the
007 * {@link GrammarBuilder} and {@link Grammar#compile compile} the resulting {@link Grammar}.
008 * </p>
009 * 
010 * @param <N> is the type of the objects created by the parser
011 * @param <C> is the type of token codes provided by the lexer
012 */
013public interface Parser<N,C extends Enum<C>> {
014  /**
015   * <p>
016   * parses a sequence of objects of type {@code <C>} from the given {@code
017   * Lexer} and transforms it into an object of type {@code <N>}.
018   * </p>
019   * 
020   * @return is a value created by a {@link NodeFactory} therefore may in
021   *         particular be {@code null}.
022   * 
023   * @throws ParseException if the sequence of {@code <C>} objects provided
024   *         by the {@code Lexer} does not match the grammar for which this
025   *         parser was created.
026   */
027  N parse(Lexer<N,C> lex) throws ParseException;
028
029}
