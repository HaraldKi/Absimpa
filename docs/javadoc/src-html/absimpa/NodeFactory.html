001package absimpa;
002
003import java.util.List;
004/**
005 * <p>
006 * creates a new {@code N} from a list of {@code N} objects provided by a
007 * parser. Whenever a parser recognizes a piece of input, the sub-pieces
008 * where already mapped to objects of type {@code N}. The list of these
009 * objects is passed by the parser to its {@link NodeFactory} to obtain a
010 * combined {@code N}.
011 * </p>
012 * 
013 * @param <N> is the type of node being created.
014 */
015
016public interface NodeFactory<N> {
017  /**
018   * <p>
019   * creates a new {@code N} from the child objects. This list of children
020   * may be empty, but it will not contain {@code null} elements.
021   * </p>
022   * 
023   * @return an {@code N} or {@code null}. A return value of {@code null}
024   *         will not be put into a list of nodes on the level above, but
025   *         will be discarded.
026   */
027  N create(List<N> children);
028}
