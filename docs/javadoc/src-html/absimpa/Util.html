001package absimpa;
002
003import java.util.Iterator;
004
005/**
006 * the typical set of static utility methods.
007 */
008public class Util {
009  private Util(){}
010  /**
011   * join a list of things into a string with a prefix, a separator and a
012   * suffix.
013   */
014  public static String join(Iterable<?> list, String prefix, String sep,
015                            String suffix)
016  {
017    StringBuilder sb = new StringBuilder();
018    sb.append(prefix);
019    Iterator<?> it = list.iterator();
020    if( it.hasNext() ) {
021      sb.append(it.next());
022    }
023    while( it.hasNext() ) {
024      sb.append(sep).append(it.next());
025    }
026    sb.append(suffix);
027    return sb.toString();
028  }
029  /*+******************************************************************/
030}
